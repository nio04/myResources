css combinator

descendent

Select all elements that are descendants of a specified element 

              
                div p {
                  /* Selects all 'p' elements that are descendants of 'div' element */
                }

child

> Select elements that are direct children of a specified element 

              
                div > p {
                 /* Selects all p' elements that are direct children of a 'div' element */
                }

adjacent

+ Selects an element that is immediately next (sibling) element by a specified element. 

              
              div + p {
               /* Selects all 'p' elements that are immediately preceded by a 'div' element */
              }

general sibling

~ Selects all elements that are siblings of a specified element. 

              
              div ~ p {
               /* Selects all 'p' elements that are siblings of a 'div' element */
              }
