firstElementChild return first child of target parent 

              
               const parent = document.querySelector("ul");
               const firstChild = parent.firstElementChild;

allTheSibling

to get all the sibling of an element, we can combine use of previousElementSibling and nextElementSibling in a function 

              
                function getAllSiblings(element) {
                const siblings = [];

                // Get all previous siblings
                const prevSibling = element.previousElementSibling;
                while (prevSibling) {
                  siblings.push(prevSibling);
                  prevSibling = prevSibling.previousElementSibling;
                }

                // Get all following siblings
                const nextSibling = element.nextElementSibling;
                while (nextSibling) {
                  siblings.push(nextSibling);
                  nextSibling = nextSibling.nextElementSibling;
                }

               return siblings;
              }

              // Example usage:
              const targetElement = document.getElementById('target');
              const allSiblings = getAllSiblings(targetElement);
              console.log(allSiblings);
